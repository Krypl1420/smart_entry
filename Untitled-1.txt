from ib_async import *

def initialize_ib(host="localhost", port=4001, client_id=1):
    ib = IB()
    ib.connect(host, port, clientId=client_id)
    return ib

def live_data(ib: IB, contract):
    """
    Subscribe to live market data for a given contract.
    """
    ticker = ib.reqMktData(contract, '', False, False)
    return ticker
